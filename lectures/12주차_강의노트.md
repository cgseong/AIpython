        'max': maximum,
        'range': maximum - minimum
    }
    
    # ì˜µì…˜ì— ë”°ë¥¸ ì¶”ê°€ ê³„ì‚°
    if options.get('include_median', False):

        if count % 2 == 0:

        result['variance'] = round(variance, 2)
        result['std_deviation'] = round(variance ** 0.5, 2)
    
    return result
